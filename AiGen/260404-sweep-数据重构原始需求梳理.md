```javascript
class ClipboardEventListener extends EventEmitter {
  startListening(dbPath) {
    // 根据平台选择对应的原生监听程序

        this.emit('change')
      }
    })
  }
}
```

## 三、与 ClipboardManager 实现对比

### 3.1 架构对比总表

二者同源架构：**渲染进程**内 `initPlugin` + **preload** 注入 Node/Electron API；数据均为 **单 JSON 文件 + `writeFileSync`** + **`fs.watch` 同步视图**。

| 维度 | ClipboardShelve | ClipboardManager |
|------|-----------------|------------------|
| **数据结构** | `data` + `collects` + `collectData` + `tags` + `tagUsage` | 仅 `data`（收藏为 `item.collect` 布尔） |
| **init过期** | `maxage == null` 时不按时间删 | 按 `setting.database.maxage` 过滤 |
| **写入防抖** | 部分操作 `queuePersist` 防抖 | 无防抖，直接 `updateDataBaseLocal` |
| **剪贴板file** | 忽略「全来自插件临时目录」的 file | 不区分临时目录 |
| **源窗口/路径** | text/image/file 可带 `sourcePaths`、窗口标题 | 无 |
| **写回剪贴板** | `restoreClipboard` + 防循环 | 无 |
| **监听降级** | 事件失败降级轮询 + 防重复注册 | `error` 时 `addCommonListener`，`close/exit` 直接 `outPlugin` |
| **UI 列表** | 懒加载 GAP、`*` 收藏搜索块、`v-memo` | 更简（无标签子系统） |
| **快捷键** | 分层 + feature 注册 + dbStorage 覆盖 | 无对等级别 |

### 4.2 推荐的数据存储方案

```javascript
// 推荐的数据存储结构设计
const DATA_SCHEMA = {
  // 索引文档 - 存储元数据和排序信息
  'clip/index': {
    _id: 'clip/index',
    version: 1,
    itemOrder: ['id1', 'id2', ...],      // 按时间排序的ID列表
    collectOrder: ['id3', 'id4', ...],   // 收藏ID列表
    tags: ['tag1', 'tag2', ...],
    tagUsage: { tag1: 5, tag2: 3 }
  },

  // 单条记录 - 每条独立存储
  'clip/item/{id}': {
    _id: 'clip/item/abc123',
    type: 'text',
    data: '文本内容',                    // 文本直接存储
    // 或 imageId: 'img/abc123',         // 图片引用附件ID
    createTime: 1712345678000,
    updateTime: 1712345678000,
    locked: false,
    collected: false,
    tags: [],
    remark: ''
  },

  // 图片附件 - 使用 postAttachment API
  'img/{id}': Buffer  // 二进制数据，最大10MB
}
```

**优势**:
- 单条更新只影响一个文档
- 图片分离存储减少主数据体积
- 支持分页加载（通过索引）
- 利用 uTools 内置云同步

### 4.3 高效实现建议总结

| 方面 | 当前实现 | 推荐方案 | 理由 |
|------|----------|----------|------|
| 数据存储 | JSON 文件 + fs | utools.db 分文档 | 文档级更新、云同步 |
| 写入策略 | 同步 + 部分防抖 | 异步 API + 批量 | 不阻塞主线程 |
| 图片处理 | Base64 内联 | 附件 API + 缩略图 | 减小数据体积 |
| 列表渲染 | v-for + v-memo | 虚拟滚动 | 万级数据流畅 |
| 批量操作 | 同步循环 | 异步分批 + 进度 | 避免页面卡死 |

## 五、当前痛点分析与解决方案

### 5.1 痛点一: ItemList 渲染性能差

#### 问题分析

1. **DOM 数量问题**: 懒加载 GAP=15，但累积后 DOM 持续增长
2. **v-memo 失效频繁**: 依赖项过多（item、locked、isActive、isSelected、isCollected）
3. **图片数据内联**: Base64 图片直接存在 DOM 属性中，大图可达数 MB
4. **响应式更新频繁**: 鼠标移动触发 activeIndex 变化

#### 解决方案: 虚拟滚动

```vue
<!-- 推荐使用 vue-virtual-scroller -->
<template>
  <RecycleScroller
    class="clip-item-list"
    :items="displayItems"
    :item-size="60"
    key-field="id"
```

### 5.2 痛点二: 数据存储方式不合理

#### 问题分析

1. **单一大文件**: 所有数据存储在一个 JSON，写入需序列化全量
2. **同步阻塞**: `writeFileSync` 阻塞主线程
3. **无索引**: 查找依赖数组遍历 O(n)
4. **图片膨胀**: Base64 编码导致体积膨胀 33%

#### 解决方案: 迁移到 uTools DB API

```javascript
// 优化后的 DB 类示例
class OptimizedDB {
  constructor() {
    this.cache = new Map()
    this.indexDoc = null
  }

  async init() {
    this.indexDoc = await utools.db.promises.get('clip/index')
    if (!this.indexDoc) {
      this.indexDoc = { _id: 'clip/index', itemOrder: [], ... }
      await utools.db.promises.put(this.indexDoc)
    }
  }

  async getItems(start = 0, limit = 50) {
    const ids = this.indexDoc.itemOrder.slice(start, start + limit)
    const docIds = ids.map(id => `clip/item/${id}`)
    return await utools.db.promises.allDocs(docIds)
  }

  async addItem(item) {
    const doc = { _id: `clip/item/${item.id}`, ...item }

    // 图片单独存储为附件
    if (item.type === 'image' && item.data) {
      const buffer = base64ToBuffer(item.data)
      await utools.db.promises.postAttachment(`img/${item.id}`, buffer, 'image/png')
      doc.imageId = `img/${item.id}`
      delete doc.data
    }

    await utools.db.promises.put(doc)

    // 更新索引
    this.indexDoc.itemOrder.unshift(item.id)
    await utools.db.promises.put(this.indexDoc)
  }
}
```

### 5.3 痛点三: 批量操作导致页面卡死

#### 问题分析

```javascript
// 当前实现 (Main.vue L683-L721)
const clearRegularTabItems = (tabType, rangeValue) => {
  const candidates = filterItemsByRange(...)
  // 同步循环删除，每条都写磁盘
  candidates.forEach((item) => {
    const ok = window.remove(item)  // 同步删除 + writeFileSync
    if (ok) removed++
  })
  handleDataRemove()  // 全量重新渲染
}
```

#### 解决方案: 异步分批处理

```javascript
const clearItemsAsync = async (items, batchSize = 50) => {
  const total = items.length
  let processed = 0
  const progress = ref({ current: 0, total })

  for (let i = 0; i < items.length; i += batchSize) {
    const batch = items.slice(i, i + batchSize)

    // 批量删除
    await db.bulkRemove(batch.map(item => item.id))

    processed += batch.length
    progress.value.current = processed

    // 让出主线程，保持 UI 响应
    await nextTick()
    await new Promise(resolve => setTimeout(resolve, 10))
  }

  handleDataRemove()
}
```

### 5.4 痛点四: 快捷键系统不完善

#### 问题分析

1. **逻辑与触发耦合**: 快捷键直接绑定 UI 操作，核心逻辑未独立
2. **自定义困难**: 修改快捷键需修改代码，无 UI 配置界面
3. **缺乏统一命令系统**: 相同操作在不同场景需重复实现

#### 解决方案: 命令模式重构

```javascript
// commands/registry.js - 统一命令系统
const commands = {
  'clipboard.copy': {
    id: 'clipboard.copy',
    label: '复制到剪贴板',
    icon: 'ContentCopy',
    execute: (ctx) => {
      copyOnly(ctx.item)
      showMessage('复制成功')
    },
    canExecute: (ctx) => ctx.item != null
  },

  'item.delete': {
    id: 'item.delete',
    label: '删除',
    execute: (ctx) => deleteItem(ctx.item),
    canExecute: (ctx) => ctx.item && !ctx.item.locked && !isCollected(ctx.item.id)
  },

  // ... 更多命令
}

// 快捷键管理器
class KeybindingManager {
  constructor() {
    this.bindings = { ...DEFAULT_BINDINGS }
    this.userBindings = utools.dbStorage.getItem('keybindings') || {}
    Object.assign(this.bindings, this.userBindings)
  }

  handleKeyDown(event, context) {
    const shortcut = eventToShortcut(event)
    const commandId = this.bindings[shortcut]
    if (commandId) {
      return executeCommand(commandId, context)
    }
    return false
  }
}

  'Escape': 'ui.escape',
  'ArrowUp': 'nav.up',
  'ArrowDown': 'nav.down',
  'Space': 'selection.toggle'
}
```
