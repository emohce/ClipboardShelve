### 2.3 剪贴板监听机制

```javascript
// listener.js - 基于原生可执行文件的监听器
class ClipboardEventListener extends EventEmitter {
  startListening(dbPath) {
    // 根据平台选择对应的原生监听程序
    const targetMap = {
      win32: 'clipboard-event-handler-win32.exe',
      linux: 'clipboard-event-handler-linux',
      darwin: 'clipboard-event-handler-mac'
    }

    // 启动子进程监听剪贴板变化
    this.child = execFile(target)

    // 当监听到 CLIPBOARD_CHANGE 事件时触发
    this.child.stdout.on('data', (data) => {
      if (data.toString().trim() === 'CLIPBOARD_CHANGE') {
        this.emit('change')
      }
    })
  }
}
```

## 3. 与 ClipboardManager 实现对比

### 3.1 架构对比

| 对比项 | ClipboardShelve (当前项目) | ClipboardManager (典型实现) |
|--------|---------------------------|----------------------------|
| **数据存储** | 本地JSON文件 + 全量读写 | uTools db API + 文档级更新 |
| **列表渲染** | 普通 v-for + v-memo | 虚拟滚动 (Virtual Scroll) |
| **批量操作** | 同步阻塞式处理 | 异步分批处理 |
| **快捷键** | 分散在组件中 | 统一命令系统 |
| **图片存储** | Base64内嵌 | 附件API分离存储 |

### 4.2 最高效实现方案

#### 4.2.1 数据存储优化方案

```javascript
// 推荐的数据存储结构设计
const DATA_SCHEMA = {
  // 索引文档 - 存储元数据和排序信息
  'clip/index': {
    _id: 'clip/index',
    version: 1,
    itemOrder: ['id1', 'id2', ...],      // 按时间排序的ID列表
    collectOrder: ['id3', 'id4', ...],   // 收藏ID列表
    tags: ['tag1', 'tag2', ...],         // 全局标签
    tagUsage: { tag1: 5, tag2: 3 }       // 标签使用统计
  },

  // 单条记录 - 每条独立存储
  'clip/item/{id}': {
    _id: 'clip/item/abc123',
    type: 'text',
    data: '文本内容',                    // 文本直接存储
    // 或 imageId: 'img/abc123',         // 图片引用附件ID
    createTime: 1712345678000,
    updateTime: 1712345678000,
    locked: false,
    collected: false,
    collectTime: null,
    tags: [],
    remark: ''
  },

  // 图片附件 - 使用 postAttachment API
  'img/{id}': Buffer  // 二进制数据
}

// 优化后的 DB 类
class OptimizedDB {
  constructor() {
    this.cache = new Map()  // 内存缓存
    this.indexDoc = null
  }

  async init() {
    // 只加载索引，延迟加载具体内容
    this.indexDoc = await utools.db.promises.get('clip/index')
    if (!this.indexDoc) {
      this.indexDoc = { _id: 'clip/index', itemOrder: [], ... }
      await utools.db.promises.put(this.indexDoc)
    }
  }

  async getItems(start = 0, limit = 50) {
    const ids = this.indexDoc.itemOrder.slice(start, start + limit)
    const docIds = ids.map(id => `clip/item/${id}`)
    const docs = await utools.db.promises.allDocs(docIds)
    return docs
  }

  async addItem(item) {
```

```javascript
// 方案 1: 异步分批处理
const clearItemsAsync = async (items, batchSize = 50) => {
  const total = items.length
  let processed = 0

  // 显示进度
  const progress = ref({ current: 0, total })

    await db.bulkRemove(batch.map(item => item.id))
```

```javascript
      const handled = executeCommand(commandId, context)
      if (handled) {
        event.preventDefault()
        event.stopPropagation()
      }
      return handled
    }

    return false
  }
}
```
